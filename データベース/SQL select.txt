--○の人の数を表示する
select count(*) as maru from s_result where icon = 'maru' and date = CURDATE()
select count(*) as batsu from s_result where icon = 'batsu' and date = CURDATE()
select count(*) as sankaku from s_result where icon = 'sankaku' and date = CURDATE()
select count(*) as comment from s_result where user_comment <> '' and date = CURDATE()

select count(icon) from s_result where icon = 'maru' and date = CURDATE();

select icon, count(icon) from s_result group by icon;

select 
( select count(*) from s_result where icon = 'maru' and date = CURDATE() group by icon ) AS maru 
, ( select count(*) from s_result where icon = 'batsu' and date = CURDATE() group by icon ) AS batsu
, ( select count(*) from s_result where icon = 'sankaku' and date = CURDATE() group by icon ) AS sankaku
,(select count(*) from s_result where user_comment <> '' and date = CURDATE() ) AS comment 
from s_result group by CURDATE();

select 
( select u.user_name from  s_result as s inner join user_login as u on s.user_id = u.user_id  where icon = 'maru' and date = CURDATE()) AS maru_user
, ( select u.user_name from  s_result as s inner join user_login as u on s.user_id = u.user_id  where icon = 'batsu' and date = CURDATE()) AS batsu_user
, (select u.user_name from  s_result as s inner join user_login as u on s.user_id = u.user_id  where icon = 'sankaku' and date = CURDATE()) AS sankaku_user
,(select u.user_name from  s_result as s inner join user_login as u on s.user_id = u.user_id where user_comment <> '' and date = CURDATE()) AS comment_user
from s_result
